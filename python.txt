import pyautogui
import time

# Interval in seconds between movements
INTERVAL = 5
# Distance to move the mouse up/down in pixels
MOVE_DISTANCE = 10

def main():
    direction = 1  # 1 = down, -1 = up
    print("Mouse movement toggler started. Press Ctrl+C to stop.")
    while True:
        try:
            x, y = pyautogui.position()
            new_y = y + (MOVE_DISTANCE * direction)
            pyautogui.moveTo(x, new_y, duration=0.1)
            direction *= -1  # Toggle direction
            time.sleep(INTERVAL)
        except KeyboardInterrupt:
            print("\nStopped by user.")
            break
        except Exception as e:
            print(f"Error: {e}")
            time.sleep(INTERVAL)

if __name__ == "__main__":
    main()
